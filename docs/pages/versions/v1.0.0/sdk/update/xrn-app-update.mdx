## 使用

<SnackInline label="Basic usage">

```tsx
import React from 'react'
import { AppUpdateChecker } from '@xrnjs/app-update'
import * as Sentry from '@sentry/react-native'

const UpdateVersionModal = () => {

	return (
		<AppUpdateChecker
			onVisibleChange={v => {
				pushAllEvent(NATIVE_EVENT_KEY.UPDATE_VERSION, { visible: v })
				if (!v) {
					hide()
				}
			}}
			noUpdate={hide}
		/>
	)
}


```
